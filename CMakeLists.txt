cmake_minimum_required(VERSION 3.7 FORCE)

# GLM dynamic library
set(GLM_DYNAMIC_LIBRARY_ENABLE OFF CACHE BOOL FORCE)

# GLM install
set(GLM_INSTALL_ENABLE ON CACHE BOOL ON FORCE)

# GLM static library
set(GLM_STATIC_LIBRARY_ENABLE OFF CACHE BOOL FORCE)

# GLM test
set(GLM_TEST_ENABLE OFF CACHE BOOL FORCE)

# Enable C++ 11
set(GLM_TEST_ENABLE_CXX_11 OFF CACHE BOOL FORCE)

# Enable C++ 14
set(GLM_TEST_ENABLE_CXX_14 OFF CACHE BOOL FORCE)

# Enable C++ 17
set(GLM_TEST_ENABLE_CXX_17 OFF CACHE BOOL FORCE)

# Enable C++ 98
set(GLM_TEST_ENABLE_CXX_98 OFF CACHE BOOL FORCE)

# Enable fast math optimizations
set(GLM_TEST_ENABLE_FAST_MATH OFF CACHE BOOL FORCE)

# Enable language extensions
set(GLM_TEST_ENABLE_LANG_EXTENSIONS OFF CACHE BOOL FORCE)

# Enable AVX optimizations
set(GLM_TEST_ENABLE_SIMD_AVX OFF CACHE BOOL FORCE)
 
# Enable AVX2 optimizations
set(GLM_TEST_ENABLE_SIMD_AVX2 OFF CACHE BOOL FORCE)

# Enable SSE2 optimizations
set(GLM_TEST_ENABLE_SIMD_SSE2 OFF CACHE BOOL FORCE)

# Enable SSE3 optimizations
set(GLM_TEST_ENABLE_SIMD_SSE3 OFF CACHE BOOL  FORCE)

# Force 'pure' instructions
set(GLM_TEST_FORCE_PURE OFF CACHE BOOL FORCE)


SUBDIRS(glm FORCE)
